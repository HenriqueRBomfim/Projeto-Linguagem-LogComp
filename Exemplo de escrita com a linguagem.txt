Campanha
criar personagem Guerreiro {
    vida = 100
    ataque = 15
}

criar monstro Orc {
    vida = 50
    ataque = 12
}

definir turno int = 1

enquanto Orc_vida > 0 {
    narre("Turno: " + turno)
    Orc_vida = Orc_vida - Guerreiro_ataque
    narre("Guerreiro ataca Orc!")

    se Orc_vida <= 0 {
        narre("Orc derrotado!")
    } senao {
        Guerreiro_vida = Guerreiro_vida - Orc_ataque
        narre("Orc ataca Guerreiro!")
    }
    
    turno = turno + 1
}
FimCampanha
